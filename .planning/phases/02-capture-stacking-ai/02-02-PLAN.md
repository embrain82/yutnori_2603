    **executeAiTurn(state): GameLogicState**
    - This is the full turn loop per D-14, D-15, D-16
    - Phase 1: THROW until throwsRemaining === 0
      ```
      while (current.turnState.throwsRemaining > 0) {
        const throwResult = generateThrow()
        current.turnState = processThrow(current.turnState, throwResult)
      }
      ```
    - Phase 2: CONSUME and APPLY each pending move
      ```
      while (current.turnState.pendingMoves.length > 0) {
        const { consumed, newTurnState } = consumeMove(current.turnState)
        current.turnState = newTurnState
        if (!consumed) break

        // Select which piece to move
        const selection = selectAiMove(current.pieces, 'ai', consumed, DEFAULT_AI_WEIGHTS)
        if (!selection) continue // no valid move, skip

        // Apply the move
        const outcome = applyMove(current.pieces, selection.pieceId, selection.moveResult)
        current.pieces = outcome.pieces

        // Handle capture extra throw (per D-01, D-03, D-16)
        if (outcome.capture.grantExtraThrow) {
          current.turnState = { ...current.turnState, throwsRemaining: current.turnState.throwsRemaining + 1 }
          // Drain extra throws
          while (current.turnState.throwsRemaining > 0) {
            const extraThrow = generateThrow()
            current.turnState = processThrow(current.turnState, extraThrow)
          }
        }

        // Handle stacking (per D-08: AI always stacks)
        if (outcome.stackOpportunity.canStack && AI_CONFIG.alwaysStack) {
          current.pieces = confirmStack(current.pieces, selection.pieceId, outcome.stackOpportunity.targetPieceId!)
        }

        // Check win condition
        const winCheck = checkWinCondition(current.pieces)
        if (winCheck.isGameOver) {
          return { ...current, isGameOver: true, winner: winCheck.winner }
        }
      }
      ```
